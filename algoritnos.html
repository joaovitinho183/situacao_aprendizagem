                2. Programas Implementados em JavaScript
                Exemplo 1: Soma de Números
                javascript
                Copiar código
                function soma(n) {
                    let soma = 0;
                    for (let i = 1; i <= n; i++) {
                        soma += i;
                    }
                    return soma;
                }
                console.log(soma(5)); // Saída: 15
                
                Exemplo 2: Verificação de Par
                javascript
                Copiar código
                function verificarPar(n) {
                    if (n % 2 === 0) {
                        return "É par";
                    } else {
                        return "É ímpar";
                    }
                }
                console.log(verificarPar(4)); // Saída: É par
                
                Exemplo 3: Cálculo de Fatorial
                javascript
                Copiar código
                function fatorial(n) {
                    if (n === 0) return 1;
                    let resultado = 1;
                    for (let i = 1; i <= n; i++) {
                        resultado *= i;
                    }
                    return resultado;
                }
                console.log(fatorial(5)); // Saída: 120
                
                
                3. Programas de Ordenação Implementados em JavaScript
                Exemplo 1: Bubble Sort
                javascript
                Copiar código
                function bubbleSort(arr) {
                    let n = arr.length;
                    for (let i = 0; i < n - 1; i++) {
                        for (let j = 0; j < n - i - 1; j++) {
                            if (arr[j] > arr[j + 1]) {
                                // Troca
                                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
                            }
                        }
                    }
                    return arr;
                }
                console.log(bubbleSort([64, 34, 25, 12, 22, 11, 90])); // Saída: [11, 12, 22, 25, 34, 64, 90]
                
                Exemplo 2: Selection Sort
                javascript
                Copiar código
                function selectionSort(arr) {
                    let n = arr.length;
                    for (let i = 0; i < n - 1; i++) {
                        let minIdx = i;
                        for (let j = i + 1; j < n; j++) {
                            if (arr[j] < arr[minIdx]) {
                                minIdx = j;
                            }
                        }
                        // Troca
                        [arr[i], arr[minIdx]] = [arr[minIdx], arr[i]];
                    }
                    return arr;
                }
                console.log(selectionSort([64, 25, 12, 22, 11])); // Saída: [11, 12, 22, 25, 64]
                
                Exemplo 3: Insertion Sort
                javascript
                Copiar código
                function insertionSort(arr) {
                    let n = arr.length;
                    for (let i = 1; i < n; i++) {
                        let key = arr[i];
                        let j = i - 1;
                        while (j >= 0 && arr[j] > key) {
                            arr[j + 1] = arr[j];
                            j--;
                        }
                        arr[j + 1] = key;
                    }
                    return arr;
                }
                console.log(insertionSort([12, 11, 13, 5, 6])); // Saída: [5, 6, 11, 12, 13]
